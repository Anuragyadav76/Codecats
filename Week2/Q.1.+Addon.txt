#include<stdio.h>
int main()
{
	int n,i,sum=0,even=0,odd=0;
	printf("Enter the number of natural numbers\n");
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		printf("%d ",i);
		sum=sum+i;
		if(i&1==1)
			odd=odd+i;
		else
			even=even+i;
	}
	printf("\nThe sum of %d natural numbers is = %d\n",n,sum);
	printf("The sum of even numbers in %d natural numbers is = %d",n,even);	//addon
	printf("\nThe sum of odd numbers in %d natural numbers is = %d",n,odd);	//addon
	getch();
	return 0;
	
}
